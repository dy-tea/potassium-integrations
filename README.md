To add to your repos add the following to `~/.config/anime-games-launcher/config.toml`:

```toml
[games]
path = "..."
registries = [
    "...",
    "https://raw.githubusercontent.com/dy-tea/potassium-integrations/refs/heads/master/games/registry.json"
]

[packages]
allow_lists = [
    "...",
    "https://raw.githubusercontent.com/dy-tea/potassium-integrations/refs/heads/master/packages/allow_list.json"
]
```
